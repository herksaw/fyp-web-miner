function switchParameter(qs, par, newPar) {
  backStr = "";
  if (qs.indexOf("&" + par + "=") != -1) {
    startPos = qs.indexOf("&" + par + "=") + 1;
    endPos = qs.indexOf("&" + par + "=") + (par.length + 1);
    if (endPos != -1) {
      backStr = qs.substring(0, startPos) + newPar + qs.substring(endPos, qs.length);
    } else {
      backStr = qs;
    }
  } else {
    backStr = qs;
  }
  return backStr;
}
